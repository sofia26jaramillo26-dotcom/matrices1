2.4. Matriz Inversa

# Calcular la matriz inversa
A = np.array([[4, 7], [2, 6]])
if np.linalg.det(A) != 0:
    A_inv = np.linalg.inv(A)
    print("Matriz A:")
    print(A)
    print("\nMatriz inversa de A:")
    print(A_inv)
    # Verificacion: A * A_inv debe ser la identidad
    identidad = np.dot(A, A_inv)
    print("\nVerificacion (A * A_inv):")
    print(np.round(identidad, 2))
else:
    print("La matriz no tiene inversa (determinante = 0)")
